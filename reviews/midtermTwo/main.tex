\begin{itemize}

\item \textbf{Organize your work}, in a reasonably neat and coherent way, in

receive very little credit.  

\item \textbf{Box your final answer} and label the solution with the appropriate variables (if applicable).

\item \textbf{Simplify your final answer}. A correct answer, written in an unsimplified way, will receive less than full credit. All fractions must be written such that the numerator and denominator share no common factors.

\begin{questions}

% Easy Points

\question[5] Write your name and the name of the instructor on page one. 
\question[5] How many points are needed to uniquely determine a line?

\addpoints
\question Use the coordinate grid to do the following:
\begin{center}
\begin{tikzpicture}[scale=0.8]

\begin{axis}[axis lines=middle,grid=both,
            xtick distance=3,ytick distance=3,minor tick num=2,
            x=0.5cm,y=0.5cm,xmin=-14,xmax=14,ymin=-8,ymax=8]
% no plots to draw ... just empty grid with axes
\end{axis}
\end{tikzpicture}
\end{center}

\begin{parts}
\part[4] Label the Four Quadrants
\part[3] Draw and Label the Points: \quad $P(6,4)$ \quad $Q(0,6)$ \quad $R(-10,-5)$
\part[3] Draw the Line through the Points: \quad $A(-5, 2)$ \quad $B(10,-4)$
\end{parts}

\addpoints
\question Find the Slope Between the Two Points:
\begin{parts}
\part[2] $$A(2,1) \text{\quad} B(3,3)$$
\vfill
\part[2] $$E(0,4) \text{\quad} F(7,5)$$
\vfill
\part[2] $$I(0,0) \text{\quad} J(2,0)$$
\vfill
\part[2] $$M(1,0) \text{\quad} N(1,9)$$
\vfill
\part[2] $$P(a,0) \text{\quad} Q(0,b)$$
\vfill
\end{parts}

% Linear Equations

\newpage

\question[5] Write the Equations:
\begin{parts}
\part Standard Form
\part Slope-Intercept
\part Point-Slope
\end{parts}

\addpoints
\question Find the Equation for the Line Between the Points in the Given Form:
\begin{parts}
\part[5]  Point-Slope: $$A(2,1) \text{\quad} B(3,3)$$
\vfill
\part[5] Slope-Intercept: $$E(0,4) \text{\quad} F(7,5)$$
\vfill
\part[5] Your Choice: $$I(0,0) \text{\quad} J(2,0)$$
\vfill
\part[5] Standard Form: $$M(1,0) \text{\quad} N(1,9)$$
\vfill
\part[5] Point-Slope: $$X(a,b) \text{\quad} Y(c,d)$$
\vfill
\end{parts}

% Matrix Stuff
\newpage

\addpoints
\question Solve the System of Linear Equations Using the Given Method:
\begin{parts}
\part[10] Substitution: $$\begin{bmatrix}
                3x - 2y = 6 \\
                2x + y = 8
                \end{bmatrix}$$
\vfill
\part[10] Elimination: $$\begin{bmatrix}
                3x + y = 4 \\
                5x + 2y = 9
                \end{bmatrix}$$
\vfill
\part[5] Your Choice: $$\begin{bmatrix}
                5x + 7y = -4 \\
                -10x + -14y = 8
                \end{bmatrix}$$
\vfill
\part[5] Your Choice: $$\begin{bmatrix}
                y = \frac{4}{3}x + 7 \\
                4x - 3y = 2
                \end{bmatrix}$$
\vfill
\end{parts}

\part[5] $$ (5x^2 - 8x + 14) - (5x^2 + 2x + 13) $$
\vfill
\part[5] $$ (x + 3) \cdot (x^2 + 5x + 6) $$
\vfill
\part[5] $$ \frac{2x^2 + 9x + 4}{x + 4} $$
\vfill
\part[5] $$ \left( \frac{16x^4y^2z^8}{12x^6y^2z^5}\right)^{\!3} $$
\vfill
\end{parts}
